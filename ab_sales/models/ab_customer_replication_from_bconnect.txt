ALTER FUNCTION [dbo].[ufn_get_tok_row]
(
    @cust_id  INT,
    @notice   NVARCHAR(MAX)
)
RETURNS TABLE
AS
RETURN
WITH Prep AS (
    SELECT notice_trim = LTRIM(RTRIM(@notice))
),
NullCheck AS (
    SELECT
        is_blank = CASE WHEN notice_trim IS NULL OR notice_trim = N'' THEN 1 ELSE 0 END,
        notice_trim
    FROM Prep
),
D1 AS (
    SELECT
        notice_trim,
        p1 = NULLIF(CHARINDEX(N'//', notice_trim), 0)
    FROM NullCheck
    WHERE is_blank = 0
),
T1 AS (
    SELECT
      name_raw = LTRIM(RTRIM(SUBSTRING(notice_trim, 1, COALESCE(p1-1, LEN(notice_trim))))),
      rest1    = CASE WHEN p1 IS NULL THEN N'' ELSE SUBSTRING(notice_trim, p1+2, LEN(notice_trim)-(p1+1)) END,
      notice_trim
    FROM D1
),
D2 AS (
    SELECT name_raw, rest1, p2 = NULLIF(CHARINDEX(N'//', rest1), 0), notice_trim FROM T1
),
T2 AS (
    SELECT
      name_raw,
      phone_raw = LTRIM(RTRIM(SUBSTRING(rest1, 1, COALESCE(p2-1, LEN(rest1))))),
      rest2     = CASE WHEN p2 IS NULL THEN N'' ELSE SUBSTRING(rest1, p2+2, LEN(rest1)-(p2+1)) END,
      notice_trim
    FROM D2
),
D3 AS (
    SELECT name_raw, phone_raw, rest2, p3 = NULLIF(CHARINDEX(N'//', rest2), 0), notice_trim FROM T2
),
Tok AS (
    SELECT
      cust_name    = name_raw,
      cust_phone   = phone_raw,   -- no normalization, just trimmed
      cust_address = LTRIM(RTRIM(SUBSTRING(rest2, 1, COALESCE(p3-1, LEN(rest2))))),
      delim_count  = (LEN(notice_trim) - LEN(REPLACE(notice_trim, N'//', N''))) / 2
    FROM D3
    UNION ALL
    -- Handle case when notice is NULL or ''
    SELECT
      cust_name    = '',
      cust_phone   = '',
      cust_address = '',
      delim_count  = 0
    FROM NullCheck
    WHERE is_blank = 1
)
SELECT
  cust_id   = @cust_id,
  cust_name,
  cust_phone ,
  cust_address,
  delim_count,
  /* First match by phone across the 3 sources (as requested) */
  --matched_cust_by_phone = (select top 1 cust_id from customer
  --where cust_mobile = cust_phone),

  /* Validation flags (BITs) */
  is_valid_name    = CASE WHEN cust_name IS NOT NULL
                               AND cust_name LIKE N'% %'
                               AND LEN(REPLACE(cust_name, N' ', N'')) >= 4
                          THEN 1 ELSE 0 END,
  is_valid_phone   = CASE WHEN dbo.ufn_is_valid_phone(cust_phone) = 1 THEN 1 ELSE 0 END,
  is_valid_address = CASE WHEN cust_address IS NOT NULL AND LEN(cust_address) >= 3
                          THEN 1 ELSE 0 END
FROM Tok;

ALTER FUNCTION [dbo].[ufn_is_valid_phone] (@raw NVARCHAR(50))
RETURNS BIT
AS
BEGIN
    IF @raw IS NULL RETURN 0;

    DECLARE @p NVARCHAR(50) = LTRIM(RTRIM(@raw));

    -- must be exactly 11 chars
    IF LEN(@p) <> 11 RETURN 0;

    -- must be digits only
    IF @p COLLATE Latin1_General_BIN LIKE '%[^0-9]%' RETURN 0;

    -- must start with 010 / 011 / 012 / 015 (3 fixed + 8 remaining)
    IF @p NOT LIKE N'01[0125]________' RETURN 0;

    RETURN 1;
END
